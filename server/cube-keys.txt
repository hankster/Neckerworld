// This is a summary of function keys interpreted by the "cube" server.
// The must be activated within the server's display window.
// They are used to test and debug server code rendering and function.
//
   A -- Activate (set velocity non-zero)
   B -- Basic strategy toggle (server controls predators resources)
   C -- Clock delay on/off
   D -- Debug toggle
   E -- Extended strategy toggle (server controls males, females, enbies)
   F -- View target -x
   G -- View target -y
   H -- View target -z
   I -- Camera Y up
   J -- Camera X down
   K -- Camera Y down
   L -- Camera Z down
   M -- Rotate cube right
   N -- Rotate cube left
   O -- Camera Z up
   P -- Screenshot
   Q -- Reset view position
   R -- View target +x
   S -- Strategy on/off
   T -- View target +y
   U -- Camera X up
   V -- Start video recording on/off
   W
   X -- Change to cube view of cube_index
   Y -- View target +z
   Z -- Increment cube index
   0-8 -- Rotate-y n x pi/4
   Left-Arrow -- Gaze left
   Right-Arrow -- Gaze right
   Up-Arrow  -- Gaze up
   Down-Arrow -- Gaze down
   Left-Bracket -- Light X -= 1.0
   Right-Bracket -- Light X += 1.0
   Period -- Light Y -= 0.5
   Slash -- Light Y += 0.5
   Minus -- Light Z -= 1.0
   Equal -- Light Z += 1.0
   Semicolon -- Ambient -= 0.1
   Apostrophe -- Ambient += 0.1
   Accent Grave -- Exit
   ESC -- Exit
   ? -- Print Help screen
